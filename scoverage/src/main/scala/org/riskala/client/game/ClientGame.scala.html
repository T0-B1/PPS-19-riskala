61    * Method that gives information about the clicked neighbor

67   @JSExport
68   def neighborClick(clickedState: String, namePlayer: String,
69                     mapSelectedState: String, gameFacade: GameFacade): Unit = {
70     if(clickedState equals mapSelectedState) {
71       gameFacade.nameActionBtn = "Deploy"
72       gameFacade.maxAvailableTroops = myTroopsToDeploy
73     } else{
74       playerStates.find(_.state == clickedState)
75         .foreach(ps => {
76           gameFacade.nameActionBtn = if (ps.owner.nickname == namePlayer) "Move" else "Attack"
77           gameFacade.maxAvailableTroops = playerStates.find(_.state == mapSelectedState).get.troops - 1
78         }
79       )
80     }
81   }

83   private def myStateInfo(playerState:PlayerState, gameFacade: GameFacade, myState: Boolean, myTurn: Boolean): Unit = {
84     gameFacade.setStateInfo(playerState.state,
85       playerState.owner.nickname,
86       playerState.troops,
87       map.regions.find(_.states.contains(playerState.state)).map(_.name).getOrElse(""))
88     val neighbors =  map.getNeighbors(playerState.state)
89     if(myState && myTurn){
90       gameFacade.visible = true
91       if(myIsDeployOnly) {
92         if(myTroopsToDeploy > 0) {
93           gameFacade.addNeighbor(playerState.state, true)
94           gameFacade.maxAvailableTroops = myTroopsToDeploy
95           gameFacade.selectedNeighbor = playerState.state
96           neighborClick(playerState.state,gameFacade.myName,playerState.state,gameFacade)
97         } else {
98           gameFacade.visible = false
99         }
100       } else {
101         if(myTroopsToDeploy > 0){
102           gameFacade.addNeighbor(playerState.state, true)
103           neighbors.foreach(gameFacade.addNeighbor(_, false))
104           gameFacade.maxAvailableTroops = myTroopsToDeploy
105           gameFacade.selectedNeighbor = playerState.state
106           neighborClick(playerState.state,gameFacade.myName,playerState.state,gameFacade)
107         } else {
108           val mySelection = neighbors.collectFirst({case s => s}).get
109           val remainingNeighbors = neighbors.filterNot(_ == mySelection)
110           gameFacade.addNeighbor(mySelection, true)
111           remainingNeighbors.foreach(gameFacade.addNeighbor(_, false))
112           gameFacade.maxAvailableTroops = playerState.troops - 1
113           gameFacade.selectedNeighbor = mySelection
114           neighborClick(mySelection,gameFacade.myName,playerState.state,gameFacade)
115         }
116       }
117     } else {
118       gameFacade.visible = false
119     }
120   }

137   @JSExport
138   def setupGame(gameInfo: String, gameFacade: GameFacade): Unit = {
139     val game = Parser.retrieveMessage(gameInfo, GameFullInfo.GameFullInfoCodecJson.Decoder).get
140     map = game.map
141     playerStates = game.playerStates
142     myTroopsToDeploy = game.troopsToDeploy
143     myActualPlayer = game.actualPlayer
144     myIsDeployOnly = game.isDeployOnly
145     game.players.foreach(pl => gameFacade.addPlayer(pl, game.actualPlayer == pl))
146     playerStates.foreach(ps => gameFacade.setPlayerState(ps.state, ps.owner.nickname, ps.troops))
147     map.regions.foreach(r => r.states.foreach(s => gameFacade.setStateRegion(s,r.name)))
148     gameFacade.setObjective(game.personalInfo.objective.info)
149     val cardOccurrence = game.personalInfo.cards.groupBy(identity).mapValues(_.size)
150     gameFacade.setCardInfo(cardOccurrence.getOrElse(Infantry, 0),
151       cardOccurrence.getOrElse(Cavalry, 0),
152       cardOccurrence.getOrElse(Artillery, 0))
153     game.winner.foreach(winner => gameFacade.setWinner(winner.nickname))
154   }

156   /**
157    * Method used to menage messages that are sent to game page
158    * */
159   @JSExport
160   def handleGameMessage(message: String, gameFacade: GameFacade): Unit = {
161     val wrappedMsg = Parser.retrieveWrapped(message).get
162     wrappedMsg.classType match {
163       case "ErrorMessage" => {
164         println("received error message")
165         val errorMsg =
166           Parser.retrieveMessage(wrappedMsg.payload, ErrorMessage.ErrorCodecJson.Decoder).get
167         gameFacade.notifyGameError(errorMsg.error)
168       }
169       case "GameUpdate" =>
170         println("received GameUpdate ")
171         val gameUpdate =
172           Parser.retrieveMessage(wrappedMsg.payload, GameUpdate.GameUpdateCodecJson.Decoder).get
173         println("Ended parser retrieve message")
174         playerStates = gameUpdate.playerStates
175         playerStates.foreach(ps => gameFacade.setPlayerState(ps.state, ps.owner.nickname, ps.troops))
176         myTroopsToDeploy = gameUpdate.troopsToDeploy
177         myActualPlayer = gameUpdate.actualPlayer
178         myIsDeployOnly = gameUpdate.isDeployOnly
179         gameFacade.maxAvailableTroops = gameUpdate.troopsToDeploy
180         gameFacade.setCurrentPlayer(gameUpdate.actualPlayer)
181         val cardOccurrence = gameUpdate.personalInfo.cards.groupBy(identity).mapValues(_.size)
182         gameFacade.setCardInfo(cardOccurrence.getOrElse(Infantry, 0),
183           cardOccurrence.getOrElse(Cavalry, 0),
184           cardOccurrence.getOrElse(Artillery, 0))
185         if(gameFacade.state!="Select a state")
186           clickedState(gameFacade.state,gameFacade.myName,gameFacade)
187 
188       case "GameEnd" =>
189         val winner =
190           Parser.retrieveMessage(wrappedMsg.payload, GameEnd.GameEndCodecJson.Decoder).get.winner
191         gameFacade.setWinner(winner.nickname)
192 
193       case "LobbyInfo" =>
194         gameFacade.goToLobby(wrappedMsg.payload)
195     }
196   }
