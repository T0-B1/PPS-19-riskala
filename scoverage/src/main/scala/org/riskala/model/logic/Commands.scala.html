62 /**
63  * An attack
64  *
65  * @param from The attacking state
66  * @param to The target state
67  * @param troops The number of troops mobilized
68  */
69 final case class Attack(from: State,
70                         to: State,
71                         troops: Int)
72                   extends SelfCheckingCommand {
73 
74   private val rng = new Random()
75   private val rndUpperBound: Int = 10
76   private val defenderThreshold: Int = 5
77 
78   override def checkedExecution(game: GameSnapshot): Behavior[Event] = {
79     val defenders = game.geopolitics.getPlayerStateByName(to).get.troops
80     var remainingDefenders = defenders
81     var remainingAttackers = troops
82     while(remainingAttackers > 0 && remainingDefenders > 0) {
83       if(rng.nextInt(rndUpperBound) > defenderThreshold)
84         remainingDefenders = remainingDefenders - 1
85       else
86         remainingAttackers = remainingAttackers - 1
87     }
88     if(remainingAttackers == 0)
89       Seq(Battle(from, to, troops, 0, defenders - remainingDefenders))
90     else{
91       val attacker = game.geopolitics.getPlayerStateByName(from).get.owner
92       val events: Seq[Event] = Seq.empty :+
93         Battle(from, to, troops, remainingAttackers, defenders) :+
94         CardDrawn(attacker, Cards.generateCard())
95       val objective: Set[State] = game.objectives(attacker).states
96       val conquered: Set[State] = game.geopolitics.getStatesOfPlayer(attacker)
97       // The ownership of the only state left is yet to be updated, but if it is the one just conquered, it's a victory
98       val diff: Set[State] = objective.diff(conquered)
99       if(diff.size == 1 && diff.head.equals(to))
100         events :+ GameEnded(attacker)
101       else
102         events
103     }
104   }
105 
106   override def feasibility(game: GameSnapshot): FeasibilityReport = {
107     val fromPS = game.geopolitics.getPlayerStateByName(from)
108     if(fromPS.isEmpty)
109       return FeasibilityReport(feasible = false, Some(s"Attacking from an unknown state: $from"))
110     val toPS = game.geopolitics.getPlayerStateByName(to)
111     if(toPS.isEmpty)
112       return FeasibilityReport(feasible = false, Some(s"Attacking an unknown state: $to"))
113     val turnFeasibility = Command.checkTurn(fromPS.get.owner, game)
114     if(!turnFeasibility.feasible)
115       return turnFeasibility
116     val availableTroops = fromPS.get.troops
117     if(availableTroops <= troops)
118       return FeasibilityReport(feasible = false, Some(s"Insufficient troops ($availableTroops) for attack with $troops from $from"))
119     FeasibilityReport()
120   }
121 }
