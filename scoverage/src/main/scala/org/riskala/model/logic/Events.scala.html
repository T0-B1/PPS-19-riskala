31 /**
32  * A battle event
33  *
34  * @param from The attacking state
35  * @param to The target state
36  * @param attacking The number of troops mobilized by the attacker
37  * @param attackingPassed The number of attacking troops that invaded the target state
38  * @param defendingCasualties The number of troops perished in the target state
39  */
40 final case class Battle(from: State,
41                        to: State,
42                        attacking: Int,
43                        attackingPassed: Int,
44                        defendingCasualties: Int)
45                   extends Event {
46   override def happen(game: GameSnapshot): GameSnapshot = {
47     var geopolitics = game.geopolitics
48     val attacker = geopolitics.getPlayerStateByName(from).get.owner
49     geopolitics = geopolitics.modifyStateTroops(from, -attacking)
50     if(attackingPassed > 0) {
51       game.copy(geopolitics = geopolitics.setStateTroops(to, attackingPassed)
52         .updateStateOwner(to, attacker))
53     } else
54       game.copy(geopolitics = geopolitics.modifyStateTroops(to, - defendingCasualties))
55   }
56 }
