29   private def lobbyManager(subscribers: Set[ActorRef[PlayerMessage]],
30                    rooms: Map[String, (ActorRef[RoomMessage], RoomBasicInfo)],
31                    games: Map[String, ActorRef[GameMessage]],
32                    terminatedGames: Map[String, (ActorRef[GameMessage], Boolean)]): Behavior[LobbyMessage] =
33     Behaviors.receive { (context, message) => {
34       
35       def nextBehavior(nextSubscribers: Set[ActorRef[PlayerMessage]] = subscribers,
36                        nextRooms: Map[String, (ActorRef[RoomMessage], RoomBasicInfo)] = rooms,
37                        nextGames: Map[String, ActorRef[GameMessage]] = games,
38                        nextTerminatedGames: Map[String, (ActorRef[GameMessage], Boolean)] = terminatedGames
39                       ): Behavior[LobbyMessage] = lobbyManager(nextSubscribers,nextRooms,nextGames,nextTerminatedGames)
40 
41       def getInfo(nextRooms: Map[String, (ActorRef[RoomMessage], RoomBasicInfo)] = rooms,
42                   nextGames: Map[String, ActorRef[GameMessage]] = games,
43                   nextTerminatedGames: Map[String, (ActorRef[GameMessage], Boolean)] = terminatedGames
44                  ): PlayerMessage = {
45         val roomList: Set[RoomNameInfo] = nextRooms.map(kv => RoomNameInfo(kv._1,
46           kv._2._2.actualNumberOfPlayer + "/" + kv._2._2.maxNumberOfPlayer)).toSet
47         val gameList: Set[String] = nextGames.keys.toSet
48         val terminatedGameList: Set[String] = nextTerminatedGames.keys.toSet
49         LobbyInfoMessage(LobbyInfo(roomList, gameList, terminatedGameList))
50       }
51 
52       def notifyAllSubscribers(info: PlayerMessage,
53                                subs: Set[ActorRef[PlayerMessage]] = subscribers): Unit = {
54         subs.foreach(s => s ! info)
55       }
56 
57       message match {
58         case Subscribe(subscriber) =>
59           subscriber ! LobbyReferent(context.self)
60           subscriber ! getInfo()
61           context.log.info(s"Subscribe from $subscriber")
62           nextBehavior(nextSubscribers = subscribers + subscriber)
63 
64         case CreateRoom(creator, roomInfo) =>
65           if(!games.contains(roomInfo.basicInfo.name) &&
66             !terminatedGames.contains(roomInfo.basicInfo.name) &&
67             !rooms.contains(roomInfo.basicInfo.name)) {
68             val room = context.spawn(RoomManager(roomInfo, context.self),
69               "RoomManager-" + roomInfo.basicInfo.name)
70             room ! Join(creator)
71             val newRooms = rooms + (roomInfo.basicInfo.name -> (room, roomInfo.basicInfo))
72             val newSubs = subscribers - creator
73             notifyAllSubscribers(getInfo(nextRooms = newRooms),newSubs)
74             nextBehavior(nextSubscribers = newSubs,nextRooms = newRooms)
75           } else {
76             creator ! ErrorMessage("Room already exists")
77             nextBehavior()
78           }
79 
80         case JoinTo(actor, name) =>
81           if(rooms.contains(name) || games.contains(name) || terminatedGames.contains(name)){
82             val newSubs = subscribers - actor
83             rooms.get(name).foreach(_._1 ! Join(actor))
84             games.get(name).foreach(_ ! JoinGame(actor))
85             terminatedGames.get(name).filter(_._2).foreach(_._1 ! JoinGame(actor))
86             nextBehavior(nextSubscribers = newSubs)
87           } else {
88             actor ! ErrorMessage("Room not found")
89             nextBehavior()
90           }
91 
92         case StartGame(info, players, roomSubscribers) =>
93           context.log.info("info -> "+info)
94           context.log.info("players -> "+players)
95           context.log.info("roomSubscribers -> "+roomSubscribers)
96           val newRooms = rooms - info.basicInfo.name
97           val game = context.spawn(GameManager(info.basicInfo.name, roomSubscribers, players,
98             players.keySet, info.scenario, context.self), "GameManager-"+info.basicInfo.name)
99           val newGames = games + (info.basicInfo.name -> game)
100           notifyAllSubscribers(getInfo(nextRooms = newRooms,nextGames = newGames))
101           nextBehavior(nextRooms = newRooms,nextGames = newGames)
102 
103         case EndGame(name, game) =>
104           val newTerminatedGames = terminatedGames + (name -> (game, true))
105           val newGames = games - name
106           notifyAllSubscribers(getInfo(nextGames = newGames,nextTerminatedGames = newTerminatedGames))
107           nextBehavior(nextGames = newGames,nextTerminatedGames = newTerminatedGames)
108 
109         case GameClosed(name, subs) =>
110           val newTerminatedGames = terminatedGames + (name -> (terminatedGames(name)._1, false))
111           val newSubs = subscribers ++ subs
112           notifyAllSubscribers(getInfo(nextTerminatedGames = newTerminatedGames),newSubs)
113           nextBehavior(nextSubscribers = newSubs,nextTerminatedGames = newTerminatedGames)
114 
115         case UpdateRoomInfo(info) =>
116           val newRooms = rooms + (info.name -> (rooms(info.name)._1, info))
117           notifyAllSubscribers(getInfo(nextRooms = newRooms))
118           nextBehavior(nextRooms = newRooms)
119 
120         case EmptyRoom(roomName) =>
121           val newRooms = rooms - roomName
122           notifyAllSubscribers(getInfo(nextRooms = newRooms))
123           nextBehavior(nextRooms = newRooms)
124 
125         case Logout(actor) => nextBehavior(nextSubscribers = subscribers - actor)
126       }
127     }
128   }
