29   private def roomPlayerBehavior(username: String,
30                                  room: ActorRef[RoomMessage],
31                                  socket: actor.ActorRef): Behavior[PlayerMessage] =
32     Behaviors.receive { (context, message) =>
33 
34     message match {
35 
36       case SocketMessage(payload) =>
37         context.log.info(s"PlayerActor of $username received socket payload: $payload")
38         val wrappedOpt = Parser.retrieveWrapped(payload)
39         if(wrappedOpt.isDefined) {
40           val wrapped = wrappedOpt.get
41           wrapped.classType match {
42 
43             case "ReadyMessage" =>
44               context.log.info("PlayerRoomActor received ReadyMessage")
45               Parser.retrieveMessage(wrapped.payload, ReadyMessage.ReadyMessageCodecJson.Decoder)
46                 .foreach(j => room ! Ready(Player(username,j.color), context.self))
47               Behaviors.same
48 
49             case "LeaveMessage" =>
50               context.log.info("PlayerRoomActor received LeaveMessage")
51               room ! Leave(context.self)
52               Behaviors.same
53 
54             case "LogoutMessage" =>
55               context.log.info("PlayerRoomActor received LogoutMessage")
56               room ! Logout(context.self)
57               Behaviors.stopped
58 
59             case "UnReadyMessage" =>
60               context.log.info("PlayerRoomActor received UnReadyMessage")
61               room ! UnReady(username, context.self)
62               Behaviors.same
63 
64             case _ =>
65               context.log.info("PlayerRoomActor received an unhandled message, IGNORED")
66               Behaviors.same
67 
68           }
69         } else {
70           context.log.info("PlayerLobbyActor failed to retrieve message, IGNORED")
71           Behaviors.same
72         }
73 
74       case RoomInfoMessage(roomInfo) =>
75         context.log.info(s"PlayerActor of $username received RoomInfoMessage")
76         socket ! TextMessage(Parser.wrap("RoomInfo",roomInfo,RoomInfo.RoomInfoCodecJson.Encoder))
77         Behaviors.same
78 
79       case LobbyReferent(lobby) =>
80         context.log.info(s"PlayerActor of $username received LobbyReferent")
81         LobbyPlayerBehavior(username,lobby,socket)
82 
83       case GameReferent(game) =>
84         context.log.info(s"PlayerActor of $username received GameReferent")
85         GamePlayerBehavior(username,game,socket)
86 
87       case RegisterSocket(newSocketActor) =>
88         context.log.info(s"PlayerActor of $username registering new socket")
89         room ! Leave(context.self)
90         room ! Join(context.self)
91         roomPlayerBehavior(username, room, newSocketActor)
92 
93       case errorMessage: PlayerMessages.ErrorMessage =>
94         context.log.info(s"PlayerActor of $username received ErrorMessage")
95         val clientError = ToClientMessages.ErrorMessage(errorMessage.error)
96         socket ! TextMessage(Parser.wrap("ErrorMessage",
97           clientError,
98           ToClientMessages.ErrorMessage.ErrorCodecJson.Encoder))
99         Behaviors.same
100 
101       case other =>
102         context.log.info(s"PlayerActor of $username received "+ other +", IGNORED")
103         Behaviors.same
104     }
105   }
