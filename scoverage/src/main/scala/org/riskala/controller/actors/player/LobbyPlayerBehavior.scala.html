32   private def lobbyPlayerBehavior(username: String,
33                           lobby: ActorRef[LobbyMessage],
34                           socket: actor.ActorRef): Behavior[PlayerMessage] = {
35     Behaviors.receive { (context,message) =>
36 
37       message match {
38 
39         case SocketMessage(payload) =>
40           context.log.info(s"PlayerActor of $username received socket payload: $payload")
41           val wrappedOpt = Parser.retrieveWrapped(payload)
42           if(wrappedOpt.isDefined) {
43             val wrapped = wrappedOpt.get
44             wrapped.classType match {
45 
46               case "JoinMessage" =>
47                 context.log.info("PlayerLobbyActor received JoinMessage")
48                 Parser.retrieveMessage(wrapped.payload, JoinMessage.JoinCodecJson.Decoder)
49                   .foreach(j => lobby ! JoinTo(context.self, j.name))
50                 Behaviors.same
51 
52               case "CreateMessage" =>
53                 context.log.info("PlayerLobbyActor received CreateMessage")
54                 Parser.retrieveMessage(wrapped.payload, CreateMessage.CreateCodecJson.Decoder)
55                   .foreach(r =>
56                     lobby ! CreateRoom(context.self, RoomInfo(RoomBasicInfo(r.name, 0, r.maxPlayer), Set.empty, r.scenario)))
57                 Behaviors.same
58 
59               case "LogoutMessage" =>
60                 context.log.info("PlayerLobbyActor received LogoutMessage")
61                 lobby ! Logout(context.self)
62                 Behaviors.stopped
63 
64               case _ =>
65                 context.log.info("PlayerLobbyActor received an unhandled message, IGNORED")
66                 Behaviors.same
67 
68             }
69           } else {
70             context.log.info("PlayerLobbyActor failed to retrieve message, IGNORED")
71             Behaviors.same
72           }
73 
74         case LobbyInfoMessage(lobbyInfo) =>
75           context.log.info(s"PlayerActor of $username received LobbyInfoMessage")
76           socket ! TextMessage(Parser.wrap("LobbyInfo",lobbyInfo,LobbyInfo.LobbyInfoCodecJson.Encoder))
77           Behaviors.same
78 
79         case errorMessage: PlayerMessages.ErrorMessage =>
80           context.log.info(s"PlayerActor of $username received ErrorMessage")
81           val clientError = ToClientMessages.ErrorMessage(errorMessage.error)
82           socket ! TextMessage(Parser.wrap("ErrorMessage",
83             clientError,
84             ToClientMessages.ErrorMessage.ErrorCodecJson.Encoder))
85           Behaviors.same
86 
87         case RegisterSocket(newSocketActor) =>
88           context.log.info(s"PlayerActor of $username registering new socket")
89           LobbyPlayerBehavior(username, lobby, newSocketActor)
90 
91         case RoomReferent(room) =>
92           context.log.info(s"PlayerActor of $username received RoomReferent")
93           RoomPlayerBehavior(username,room,socket)
94 
95         case GameReferent(game) =>
96           context.log.info(s"PlayerActor of $username received GameReferent")
97           GamePlayerBehavior(username,game,socket)
98 
99         case other =>
100           context.log.info(s"PlayerActor of $username received "+ other +", IGNORED")
101           Behaviors.same
102 
103       }
104     }
105   }
