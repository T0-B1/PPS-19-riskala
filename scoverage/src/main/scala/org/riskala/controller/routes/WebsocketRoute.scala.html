25   val websocketRoute: Route =
26     (get & path("websocket") & parameter("token")) { token =>
27       if( (!AuthManager.checkToken(token)) || AuthManager.getUserName(token).isEmpty) complete(StatusCodes.Forbidden)
28       else{
29         val username = AuthManager.getUserName(token).get
30         handleWebSocketMessages(createSocketFlow(username))
31       }
32     }
33 
34   private def createSocketFlow(username: String): Flow[Message, Message, NotUsed] = {
35     val (wsActor, wsSource) = untypedActorSource().preMaterialize()
36     val playerActorRef: ActorRef[PlayerMessage] = spawnOrGetPlayer(username, wsActor)
37     Flow.fromSinkAndSource(sinkFromActor(playerActorRef), wsSource)
38   }
39 
40   /**
41    * Retrieves the PlayerActor of a particular user and returns it.
42    * If absent, a new one is generated and returned.
43    *
44    * @param username The username
45    * @param newSocketActor The actor encapsulating the websocket
46    * @return
47    */
48   private def spawnOrGetPlayer(username: String, newSocketActor: actor.ActorRef): ActorRef[PlayerMessage] = {
49     Utils.askReceptionistToFind[PlayerMessage](username).toSeq match {
50       // No PlayerActor registered found
51       case Seq() =>
52         val ref: ActorRef[PlayerMessage] = system.systemActorOf(PlayerActor(username, newSocketActor), username)
53         system.receptionist ! Receptionist.Register(ServiceKey[PlayerMessage](username), ref)
54         ref
55 
56       // The first PlayerActor found
57       case Seq(first, rest@_*) =>
58         first ! PlayerMessages.RegisterSocket(newSocketActor)
59         first
60 
61     }
62   }
63 
64   private def typedActorSource(): Source[Message, ActorRef[Message]] = ActorSource.actorRef[Message](completionMatcher = {
65     case _ => CompletionStrategy.immediately
66   }, PartialFunction.empty, bufferSize = 8, overflowStrategy = OverflowStrategy.dropHead)
67 
68   private def untypedActorSource(): Source[Message, actor.ActorRef] =
69     Source.actorRef[Message](bufferSize = 8, overflowStrategy = OverflowStrategy.dropHead)
70 
71   private def sinkFromActor(playerRef: ActorRef[PlayerMessage]): Sink[Message, Future[Done]] =
72     Sink.foreach(m => playerRef ! PlayerMessages.SocketMessage(m.asTextMessage.getStrictText))
