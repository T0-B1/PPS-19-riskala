35   /**
36    * Asks the akka system receptionist to find an actor
37    *
38    * @param key ServiceKey of the actor
39    * @param system Akka actor system
40    * @param timeout Timeout of the request
41    * @tparam T Type of the ActorRef
42    * @return An optional ActorRef
43    */
44   def askReceptionistToFind[T:ClassTag](key: String)(implicit system: ActorSystem[Nothing], timeout: Timeout = 3.seconds) : Set[ActorRef[T]] = {
45     implicit val executionContext: ExecutionContextExecutor = system.executionContext
46     val serviceKey: ServiceKey[T] = ServiceKey[T](key)
47     val receptionist: ActorRef[Receptionist.Command] = system.receptionist
48     val response: Future[Receptionist.Listing] =
49       receptionist ? (replyTo => Receptionist.find(serviceKey, replyTo))
50     Await.result(response, 3.seconds).serviceInstances(serviceKey)
51   }
